// src/utils/CarbonContext.ts

const DAILY_CARBON_ALLOWANCE_GRAMS = 6300; // The 1.5Â°C Goal

export function getSovereigntyScore(sessionGramsCO2: number) {
  const percentUsed = (sessionGramsCO2 / DAILY_CARBON_ALLOWANCE_GRAMS) * 100;
  
  let verdict = "Sustainable";
  let color = "green";

  if (percentUsed > 5) {
    verdict = "Heavy Load";
    color = "yellow";
  }
  if (percentUsed > 25) {
    verdict = "Moltbook Level (Wasteful)";
    color = "red";
  }

  return {
    percentOfHumanQuota: percentUsed.toFixed(4) + "%",
    verdict,
    color
  };
}
